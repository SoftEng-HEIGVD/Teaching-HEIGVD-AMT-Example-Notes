390      $.fn.selectpicker = function(option, event) {
391         //get the args of the outer function..
392         var args = arguments;
393         var value;
394         var chain = this.each(function () {
395              var $this = $(this),
396                  data = $this.data('selectpicker'),
397                  options = typeof option == 'object' && option;
398              
399              if (!data) {
400              	$this.data('selectpicker', (data = new Selectpicker(this, options, event)));
401              } else {
402              	for(var i in option) {
403              		data[i]=option[i];
404              	}
405              }
406              
407              if (typeof option == 'string') {
408                  //Copy the value of option, as once we shift the arguments
409                  //it also shifts the value of option.
410                  property = option;
411                  if(data[property] instanceof Function) {
412                      [].shift.apply(args);
413                      value = data[property].apply(data, args);
414                  } else {
415                      value = data[property];
416                  }
417              }
418          });
419          
420          if(value!=undefined) {
421              return value;
422          } else {
423              return chain;
424          } 
425      };
