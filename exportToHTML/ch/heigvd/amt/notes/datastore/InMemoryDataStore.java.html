1    package ch.heigvd.amt.notes.datastore;
2    
3    import ch.heigvd.amt.notes.datastore.exceptions.DuplicateKeyException;
4    import ch.heigvd.amt.notes.datastore.exceptions.KeyNotFoundException;
5    import ch.heigvd.amt.notes.model.Note;
6    import ch.heigvd.amt.notes.model.User;
7    
8    import javax.ejb.Singleton;
9    import java.util.List;
10   import java.util.concurrent.ConcurrentHashMap;
11   import java.util.stream.Collectors;
12   
13   @Singleton
14   public class InMemoryDataStore implements IInMemoryDatastore {
15   
16     public static final boolean INCLUDE_DELETED = true;
17   
18     /* 
19      * We store users and notes in concurrent hashmaps, with values indexed 
20      * by unique ids (username for Users, id for Notes). 
21      * 
22      * In the methods, you will see that we clone objects (on the way in and out of the 
23      * data store). The reason is that we don't want a client to pass an object to the 
24      * data store (by reference), modify it on its side, and have an impact on the object 
25      * under the control of the data store. Since our Model objects are immutable, this 
26      * would not be strictly necessary. 
27      */
28     ConcurrentHashMap<String, User> storeUsers = new ConcurrentHashMap<>();
29     ConcurrentHashMap<Long, Note> storeNotes = new ConcurrentHashMap<>();
30   
31     /* 
32      * This variable is used to allocate unique IDs to notes 
33      */
34     long lastNoteId = 0;
35   
36     @Override
37     public List<User> getAllUsers() {
38       return storeUsers
39         .values()
40         .stream()
41         .map(user -> {
42           User clone = user.toBuilder().build();
43           return clone;
44         }).collect(Collectors.toList());
45     }
46   
47     @Override
48     public void insertUser(User user) throws DuplicateKeyException {
49       if (storeUsers.get(user.getUsername()) != null) {
50         throw new DuplicateKeyException("User with username " + user.getUsername() + " already exists.");
51       }
52       User clone = user.toBuilder().build();
53       storeUsers.put(user.getUsername(), clone);
54     }
55   
56     @Override
57     public User loadUserByUsername(String username) throws KeyNotFoundException {
58       User user = storeUsers.get(username);
59       if (user == null) {
60         throw new KeyNotFoundException("Could not find user " + username);
61       }
62       User clone = user.toBuilder().build();
63       return clone;
64     }
65   
66     @Override
67     public void updateUser(User user) throws KeyNotFoundException {
68       User storedUser = storeUsers.get(user.getUsername());
69       if (storedUser == null) {
70         throw new KeyNotFoundException("Could not find user wit username " + user.getUsername());
71       }
72       User clone = user.toBuilder().build();
73       storeUsers.put(user.getUsername(), clone);
74     }
75   
76     @Override
77     public Note getNoteById(long noteId) throws KeyNotFoundException {
78       Note storedNote = storeNotes.get(noteId);
79       if (storedNote == null) {
80         throw new KeyNotFoundException("Could not find note wit id " + noteId);
81       }
82       return storedNote.toBuilder().build();
83     }
84   
85     @Override
86     public List<Note> getNotesByAuthorUsername(String username, boolean includeDeleted) throws KeyNotFoundException {
87       if (storeUsers.get(username) == null) {
88         throw new KeyNotFoundException("Could not find user " + username );
89       }
90       List<Note> userNotes = storeNotes.values()
91         .stream()
92         .filter(note -> note.getAuthor().getUsername().equals(username))
93         .filter(note -> includeDeleted || !note.isDeleted())
94         .collect(Collectors.toList());
95   
96       return userNotes
97         .stream()
98         .map(note -> note.toBuilder().build())
99         .collect(Collectors.toList());
100    }
101  
102    @Override
103    public synchronized long insertNote(String username, Note note) throws KeyNotFoundException {
104      if (storeUsers.get(username) == null) {
105        throw new KeyNotFoundException("Could not find user " + username );
106      }
107      User authorStoredInDB = storeUsers.get(username).toBuilder().build();
108      lastNoteId++;
109      Note storedNote = note.toBuilder()
110        .id(lastNoteId)
111        .author(authorStoredInDB)
112        .build();
113      storeNotes.put(lastNoteId, storedNote);
114      return lastNoteId;
115    }
116  
117    @Override
118    public void updateNote(Note note) throws KeyNotFoundException {
119      Note storedNote = storeNotes.get(note.getId());
120      if (storedNote == null) {
121        throw new KeyNotFoundException("Could not find note with id " + note.getId());
122      }
123      storeNotes.put(note.getId(), note.toBuilder().build());
124    }
125  
126    @Override
127    public void deleteNote(long noteId) throws KeyNotFoundException {
128      Note storedNote = storeNotes.get(noteId);
129      if (storedNote == null) {
130        throw new KeyNotFoundException("Could not find note with id " + noteId);
131      }
132      Note deletedNote = storedNote.toBuilder().isDeleted(true).build();
133      updateNote(deletedNote);
134    }
135  }
